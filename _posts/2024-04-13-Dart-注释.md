```dart
   * Symbol
   *
  Symbol libraryName = new Symbol('dart.core');
  MirrorSystem mirrorSystem = currentMirrorSystem();
  LibraryMirror libMirror = mirrorSystem.findLibrary(libraryName);
  libMirror.declarations.forEach((s, d) => print('$s - $d'));
  */
```
